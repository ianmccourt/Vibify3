// Run this in your browser console when on Vibify
const tabs = document.querySelectorAll('button');
console.log('Total buttons:', tabs.length);
tabs.forEach(btn => {
    const text = btn.textContent?.trim();
    if (text && text.length < 30) {
        const rect = btn.getBoundingClientRect();
        console.log(`"${text}" - top:${rect.top}px, visible:${rect.height > 0}`);
    }
});
